									parnet.prepend($elem) // 부모가 자식을 추가함
									parnet.prepend(elem) // 복수 노드 추가 가능

									// 추가 (처음2)
									$elem.prependTo(parnet) // 자식이 부오에 추가됨
									parnet.prependChild(elem) // 하나의 노드만 추가 가능

									// 추가 (마지막1)
									parnet.append($elem) // 부모가 자식을 추가함
									parnet.append(elem) // 복수 노드 추가 가능

									// 추가 (마지막2)
									$elem.appendTo(parnet) // 자식이 부오에 추가됨
									parnet.appendChild(elem) // 하나의 노드만 추가 가능

									// 추가 (이전)

									A.insertBefore(B) // B 앞에 A를 추가
