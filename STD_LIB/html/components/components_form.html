                                    fileAttach: function(obj, e){
                                        // 파일첨부 - 파일명적용
                                        var sel_files = [];
                                        var $eleFormText = $(obj).closest('.form-file').find('input[type=text]');
                                        var $eleFormImg = $(obj).closest('.form-file').find('.file-preview');

                                        var action = function(){
                                            //파일명 적용
                                            if ($eleFormText.length){
                                                var fileValue = $(obj).val().split("\\");
                                                var fileName = fileValue[fileValue.length-1];
                                                $eleFormText.val(fileName);
                                            }

                                            //이미지 미리보기
                                            if ($eleFormImg.length){
                                                var files = e.target.files;
                                                var filesArr = Array.prototype.slice.call(files);

                                                filesArr.forEach(function(f){
                                                    if (!f.type.match("image.*")){
                                                        alert('확장자는 이미지 확장자만 가능합니다.');
                                                        return;
                                                    }
                                                    sel_files.push(f);

                                                    var reader = new FileReader();
                                                    reader.onload = function(e){
                                                        var eleImg = '';
                                                        $eleFormImg.html(eleImg);
                                                    }
                                                    reader.readAsDataURL(f);
                                                })
                                            }
                                        }
                                        action();
                                    },
설계가이드
  • 묵시적인 방법(label 에 checkbox, radio 포함)은 일부 스크린리더에서 해당 마크업을 제대로 읽지 못하므로
  • 명시적인 방법(input id 와 label for 연결)으로 사용하는 것을 권장합니다.
