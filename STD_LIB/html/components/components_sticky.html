										init: function(){
											this.action();
											this.event();
										},
										event: function(){
											var setTime;
											$(window).on('scroll', function(){

											else { $('body').removeClass('is-scrolled') }
										}
									}

Sticky Scroll

Single Sticky

HTML CSS SCRIPT


[Sticky 새창열기] 
									<div class="sticky-scroll-area">
										<div class="tab-nav tab-basic sticky-scroll"></div>
										<div class="sticky-scroll-after"></div>
									</div>
								
									/* Common - Scroll */
									.sticky-scroll-area {position:relative;}
									.sticky-scroll {position:absolute; left:0; top:0; width:100%;}
									.sticky-scroll.is-fixed {position:fixed;}
									.sticky-scroll-after {display:block;}

									/* UI - Sticky */
									.tab-basic.sticky-scroll + .sticky-scroll-after {display:block; height:35px;}
								
									var ui = {
										init: function(){
											if ($('.tab-nav.sticky-scroll').length){ ui.stickyScroll.init() }
										},
										stickyScroll: {
											eleModule: '.sticky-scroll',
											eleWrapper: '.sticky-scroll-area',
											init: function(){
												this.action();
												this.event();
											},
											event: function(){
												$(window).on('scroll', function(){
													ui.stickyScroll.action();
												})
											},
											action: function(){
												var _this = this;
												$(this.eleModule).each(function(){
													var objTop = $(this).closest(_this.eleWrapper).offset().top;
													var scrTop = $(window).scrollTop();
													if (scrTop >= objTop){ $(this).addClass('is-fixed') }
													else { $(this).removeClass('is-fixed') }
												})
											}
										}
									}
									$(document).ready(function(){
										ui.init();
									})

Multi Sticky

HTML CSS SCRIPT


[Sticky 새창열기] 
									<div class="sticky-scroll-area">
										<div class="tab-nav tab-basic sticky-scroll"></div>
										<div class="sticky-scroll-after"></div>
									</div>
								
									/* Common - Scroll */
									.sticky-scroll-area {position:relative;}
									.sticky-scroll {position:absolute; left:0; top:0; width:100%;}
									.sticky-scroll.is-fixed {position:fixed;}
									.sticky-scroll-after {display:block;}

									/* UI - Sticky */
									.tab-basic.sticky-scroll + .sticky-scroll-after {display:block; height:35px;}
								
									var ui = {
										init: function(){
											if ($('.tab-nav.sticky-scroll').length){ ui.stickyScroll.init() }
										},
										stickyScroll: {
											eleModule: '.sticky-scroll',
											eleWrapper: '.sticky-scroll-area',
											init: function(){
												this.action();
												this.event();
											},
											event: function(){
												$(window).on('scroll', function(){
													ui.stickyScroll.action();
												})
											},
											action: function(){
												var _this = this;
												$(this.eleModule).each(function(){
													var $eleModule = $(this);
													var objTop = $eleModule.closest(_this.eleWrapper).offset().top;
													var scrTop = $(window).scrollTop();

													//기존 StickyFixed 가 있는 경우
													var fixedH;
													$('.sticky-fixed').each(function(){
														fixedH = $(this).outerHeight();
														objTop = objTop - fixedH;
													})

													if (scrTop >= objTop){ $eleModule.addClass('is-fixed'); $eleModule.css({top: fixedH}); }
													else { $eleModule.removeClass('is-fixed'); $eleModule.css({top: 0}); }

													console.log(objTop, scrTop);
												})
											}
										}
									}
									$(document).ready(function(){
										ui.init();
									})

Sticky Floating

Basic

HTML CSS SCRIPT


[Sticky 새창열기] 
									<div class="pageTop sticky-float"></div>
								
									.pageTop {position:absolute; right:20px; top:calc(100% - 70px);}
								
									stickyFloat: {
										eleModule: '.sticky-float',
										init: function(){
											this.action();
											this.event();
										},
										event: function(){
											var setTime;
											$(window).on('scroll', function(){
												clearTimeout(setTime);
												setTime = setTimeout(function(){ ui.stickyFloat.action(); }, 100);
											})
										},
										action: function(){
											var scrTop = $(window).scrollTop() - $(document).height() + $(window).outerHeight();
											TweenMax.to($(this.eleModule), 0.5, {y: scrTop, ease: Power2.easeOut})
										}
									}

Limit

SCRIPT


[Sticky 새창열기] 
									stickyFloat: {
										eleModule: '.sticky-float',
										init: function(){
											this.action();
											this.event();
										},
										event: function(){
											var setTime;
											$(window).on('scroll', function(){
												clearTimeout(setTime);
												setTime = setTimeout(function(){ ui.stickyFloat.action(); }, 100);
											})
										},
										action: function(){
											var scrTop = $(window).scrollTop() - $(document).height() + $(window).outerHeight() + 70;

											// Footer가 보이는 경우
											var footerTop = $('.footer').offset().top;
											if (footerTop <= $(window).scrollTop() + $(window).height()){ scrTop = 0; }
											TweenMax.to($(this.eleModule), 0.5, {y: scrTop, ease: Power2.easeOut});
										}
									}
