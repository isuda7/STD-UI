HTML

기본구조

<!-- Swiper --> <div class="swiper_demo"> <!-- Swiper Controls --> <button type="button" class="swiper-button-prev"><span>이전</span></button> <button type="button" class="swiper-button-next"><span>다음</span></button> <div class="swiper-autoplay"> <button type="button" class="swiper-autoplay-start"><span>자동재생 시작</span></button> <button type="button" class="swiper-autoplay-stop"><span>자동재생 정지</span></button> </div> <div class="swiper-pagination"></div> <!-- // Swiper Controls --> <!-- Swiper Container --> <div class="swiper-container"> <ul class="swiper-wrapper"> <li class="swiper-slide"></li> <li class="swiper-slide"></li> <li class="swiper-slide"></li> <li class="swiper-slide"></li> </ul> </div> <!-- // Swiper Container --> </div> <!-- // Swiper -->
  • .swiper_이름 > 컨트롤러 + 컨테이너 구조가 되어야 한다.
  • 컨트롤러가 컨테이너 내부에 있을경우, 밖으로 표현할 수 없음.
  • 컨트롤러가 컨테이너보다 앞에 있는것이 올바른 구조이나, 상황에 따라 변경 가능하다. (인증마크인 경우 변경불가)

Script

호출방법

var title = '스와이프제목'; var $swiper = $('.swiper_demo'); var $container = $swiper.find('>.swiper-container'); var options = {} var func = function(options){ demoSwiper = new Swiper($container, options); } ui.swipes.validate($container, title, options, func);
  • API 참고 페이지로 이동
  • ui.swipes.validate($container, title, options, func)를 통해서 실행한다.
  • 예) demoSwiper는 전역변수에 선언하여, 어디에서든 제어할 수 있어야 한다.

실행공통

// ui.swipes.validate() validate: function($container, title, options, func){ var $swiper = $container.parent(); var $navDisabled = $swiper.children('.swiper-button-disabled'); if (title == undefined){ title = '' } var defaultOptions = { watchSlidesVisibility: true, a11y: { prevSlideMessage: title+ ' 이전 슬라이드', nextSlideMessage: title+ ' 다음 슬라이드', firstSlideMessage: title+ ' 첫번째 슬라이드', lastSlideMessage: title+ ' 마지막 슬라이드', }, on: { init: function(){ ui.setSwiper($container, 'swiper'); }, slideChangeTransitionStart: function(){ ui.setSwiper($container, 'swiper'); } } } $.extend(defaultOptions, options); // 기본옵션 머지하기 //몇개부터 스와이프 설정할지 체크 if ($container.find('>.swiper-wrapper>.swiper-slide:visible').length > 1){ // 보이는 슬라이드가 1개 초과일때 func(defaultOptions); // 스와이프 호출(옵션보내기) $swiper.removeClass('no_swiper').addClass('is_swiper'); // 스와이프 적용클래스 if (defaultOptions.autoplay != undefined && defaultOptions.autoplay != false){ $swiper.addClass('is_started').removeClass('is_stoped') } // 자동재생인 경우 재생상태 적용 if ($navDisabled.length == 2){ $navDisabled.addClass('dis_n') } // 양쪽 disabled 인경우 숨김 else if ($navDisabled.length){ $navDisabled.removeClass('dis_n') } // 한쪽 disabled 인경우 초기화 } else { $swiper.removeClass('is_swiper').addClass('no_swiper'); } },

참고사항

var options = { // 참고옵션 (navigation 및 pagination 클래스명은 기본 클래스명으로 사용합니다.) navigation: { nextEl: $swiper.children('.swiper-button-next'), //다음 슬라이드 prevEl: $swiper.children('.swiper-button-prev'), //이번 슬라이드 }, pagination: { el: $swiper.children('.swiper-pagination'), type: 'bullets', renderBullet: function (index, className) { return '<span class="' + className + '">' + (index + 1) + '번 슬라이드</span>'; } }, } var func = function(options){ demoSwiper = new Swiper($container, options); // 참고이벤트 $document.off('click.demoSwiperStart').on('click.demoSwiperStart', '.swiper_demo>.swiper-autoplay>.swiper-autoplay-start', function(){ demoSwiper.autoplay.start(); $swiper.addClass('is_started').removeClass('is_stoped'); }); $document.off('click.demoSwiperPause').on('click.demoSwiperPause', '.swiper_demo>.swiper-autoplay>.swiper-autoplay-stop', function(){ demoSwiper.autoplay.stop(); $swiper.addClass('is_stoped').removeClass('is_started'); }); }
  • 컨트롤을 컨테이너 밖으로 꺼내는 경우, 선택자를 $swiper부터 잡아준다.

CSS 기본값

.swiper-button-next, .swiper-button-prev, .swiper-pagination {display:none;} .is_swiper>.swiper-button-next, .is_swiper>.swiper-button-prev, .is_swiper>.swiper-pagination {display:block;}
  • 컨트롤러는 기본값으로 숨김처리 하며, 스와이프가 설정된 경우 is_swiper 클래스로 노출한다.
  • 스크립트가 로딩되기까지 불필요하게 노출되는 경우를 대응한다.
