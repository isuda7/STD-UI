CSS SCRIPT 
									/* Effect Wave */
									.effect-wave {display:block; position:absolute; transform:translateX(-50%) translateY(-50%); width:0px; height:0px; background-color:#000; border-radius:50%; opacity:0.2;}
								
									waveEffect : {
										init : function(){
											this.event();
										},
										event : function(){
											/* Add the class effect on button */
											$('.button-effect').not('.not-effect').on('click', function(e) {
											  var self = $(this),
												  wave = '.effect-wave',
												  /* Get the width of button (if different buttons types) */
												  btnWidth = self.outerWidth(),
												  x = e.offsetX,
												  y = e.offsetY;

											  self.prepend('');

											  $(wave)
												.css({'top': y, 'left': x})
												.animate({opacity: '0', width: btnWidth * 2.2, height: btnWidth * 2.2 }, 500, function() {
													self.find(wave).remove();
												})
											})
										}
									},
