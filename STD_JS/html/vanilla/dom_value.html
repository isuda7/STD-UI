Size

									// jQuery
									$elem.width() 		// 요소의 폭만
									$elem.innerWidth() 	// 요소의 padding을 포함한 폭
									$elem.outerWidth() 	// 요소의 border, padding을 포함한 폭
									$elem.outerWidth(true) //  요소의 margin, border, padding을 포함한 폭

									$elem.height() 		// 요소 높이만
									$elem.innerHeight() // 요소의 padding을 포함한 높이
									$elem.outerHeight() // 요소의 border, padding 포함한 높이
									$elem.outerHeight(true) // 요소의 margin, border, padding 포함한 높이

									$(width).width()	   // 스크롤 바를 포함하지 않은 너비
									$(width).outerWidth()  // 스크롤 바를 포함한 너비
									$(width).height()	   // 스크롤 바를 포함하지 않은 너비
									$(width).outerHeight() // 스크롤 바를 포함한 너비

									// Varnilla
									$elem.clientWidth	// padding을 포함한 폭
									$elem.offsetWidth	// border, padding, 스크롤 바를 포함한 폭
									$elem.scrollWidth	// padding을 포함한 스크롤 폭

									$elem.clientHeight	// padding을 포함한 높이
									$elem.offsetHeight	// border, padding, 스크롤 바를 포함한 높이
									$elem.scrollHeight	// padding을 포함한 스크롤 높이

									window.innerWidth	// 스크롤 바를 포함한 창 너비
									window.innerHeight	// 스크롤 바를 포함한 창 높이
									window.outerWidth	// 스크롤 바와 창틀을 포함한 창의 너비
									window.outerHeight	// 스크롤 바와 창틀을 포함한 창의 높이

Event 좌표

									// 화면 기준
									1. screenX, screenY
									사용자 모니터 화면을 기준으로 한 좌표를 표시합니다.
									사용자 모니터의 왼쪽 상단 모서리가 (0, 0)이 됩니다.
									 
									// 문서 기준
									2. pageX, pageY
									전체 문서를 기준으로 한 좌표를 표시합니다.
									스크롤시 좌표값은 변경됩니다.
									 
									// 웹뷰 기준
									3. clientX, clientY
									브라우저에서 사용자에게 웹페이지가 보여지는 영역을 기준으로 좌표를 표시합니다.
									스크롤시 좌표값은 동일합니다.
									 
									// Dom 기준
									4. offsetX, offsetY
									이벤트가 걸려있는 DOM node를 기준으로 좌표를 표시합니다.
