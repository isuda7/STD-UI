Elements

									// body
									$("body")
									document.body

									// html
									$("html")
									document.documentElement

									// tag
									$("selector")
									document.getElementsByClassName('selector')

									// id
									$("#selector")
									document.getElementById('selector')

									// class 단일
									$(".selector")
									document.querySelector('.selector')
									>> 반환객체는 한개의 요소만 찾으며, 여러개일 경우 첫번째 요소를 반환합니다.

									// class 복수
									$(".selector")
									document.querySelectorAll('.selector')
									>> 반환객체는 nodeList이기 때문에 for문 또는 foreach 문을 사용해야 합니다.

									// Varialble
									const $ele = $(selector)
									const ele = document.querySelector(selector)

									// 자식
									$ele.children()
									ele.children

									// 부모
									$ele.parent()
									ele.parentNode

									// 가까운 상위
									$ele.closest('.country')
									ele.closest('.country')

									// 가까운 상위 폼
									$input.closest('form')
									input.form

									// 이전
									$ele.prev()
									ele.previousElementSibling

									// 다음
									$ele.next()
									ele.nextElementSibling

									// 형제요소
									$ele.siblings()
									const siblings = (el) => {
										return  Array.prototype.filter.call(el.parentElement.children, function(e){ return e !== el;});
									}
									let sbl = siblings(el);

									$ele:eq(n)
